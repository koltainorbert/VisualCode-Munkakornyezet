naplo-watcher.ps1

percenkénti auto-push

VS Code háttérben fut. Változás esetén updateli NAPLO.md-t és pusholja GitHubra.

.vscode/naplo-watcher.ps1
$temaMappa = "D:\LocalWP\sdh2026\app\public\wp-content\themes\bricks-child"
$naplo = "$temaMappa\NAPLO.md"
Set-Location $temaMappa
Write-Host "Naplo watcher elindult..." -ForegroundColor Green
$utolso = ""
while ($true) {
    $most = Get-Date
    $datumSor = $most.ToString("yyyy. MMMM dd.")
    $ido = $most.ToString("HH:mm")
    $valtozott = git diff --name-only HEAD 2>$null
    $untracked = git ls-files --others --exclude-standard 2>$null
    $osszes = @($valtozott) + @($untracked) | Where-Object { $_ -ne "" }
    if ($osszes.Count -gt 0) {
        $bejegyzes = "$ido - modositva: $($osszes -join ', ')"
        if ($bejegyzes -ne $utolso) {
            $utolso = $bejegyzes
            git add .
            git diff --cached --quiet
            if ($LASTEXITCODE -ne 0) {
                git commit -m "Auto-mentes $ido"
                git push
                Write-Host "[$ido] Feltoltve" -ForegroundColor Green
            }
        }
    } else { Write-Host "[$ido] Nincs valtozas." -ForegroundColor DarkGray }
    Start-Sleep -Seconds 60
}

deploy.yml

GitHub Actions — FTP auto-deploy

Main push → éles szerver. Helye: .github/workflows/deploy.yml

.github/workflows/deploy.yml
name: Deploy - Éles szerver

on:
  push:
    branches: [ main ]

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4
      - name: FTP Deploy
        uses: SamKirkland/FTP-Deploy-Action@v4.3.4
        with:
          server: ${{ secrets.FTP_SERVER }}
          username: ${{ secrets.FTP_USERNAME }}
          password: ${{ secrets.FTP_PASSWORD }}
          server-dir: ${{ secrets.FTP_SERVER_DIR }}
